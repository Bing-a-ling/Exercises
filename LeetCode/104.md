递归 

树的最大深度等于子树深度最大值加一

空指针深度为0
```
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxDepth(TreeNode* root) {
        if(root == nullptr)
            return 0;
        int da = maxDepth(root->left);
        int db = maxDepth(root->right);
        return max(da,db) + 1;
    }
};
```
