class Solution {
public:
    int missingNumber(vector<int>& nums) {
        vector<int> show(nums.size()+1, false);
        int res;
        for(int i = 0; i < nums.size(); i++){
            show[nums[i]] = true;
        }
        for(int i = 0; i <= nums.size(); i++){
            if(!show[i]){
                res = i;
                break;
            };
        }
        return res;
    }
};
