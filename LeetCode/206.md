class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        stack<ListNode*> Stack;
        if(!head) return nullptr;
        while(head){
            Stack.push(head);
            head = head->next;
        }
        ListNode* newHead = Stack.top(), *thePre = newHead, *theNext;
        Stack.pop();
        while(!Stack.empty()){
            theNext = Stack.top();
            Stack.pop();
            thePre->next = theNext;
            thePre = theNext;
        }
        theNext->next = NULL;
        return newHead;
    }
};
