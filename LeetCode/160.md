将两链表分别入栈

同时出栈两链表中指针直到指针不相同

最后一个相同指针即为所求
```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        stack<ListNode*> listA;
        stack<ListNode*> listB;
        ListNode* pA = headA, *pB = headB;
        while(pA){
            listA.push(pA);
            pA = pA->next;
        }
        while(pB){
            listB.push(pB);
            pB = pB->next;
        }
        ListNode* res = nullptr;
        while(!listA.empty() && !listB.empty() && listA.top() == listB.top()){
            res = listA.top();
            listA.pop();listB.pop();
        }
        return res;
    }
};
```

可以将A尾和B头连起来，判断新的长链中有没有环 之后寻找环的入口

使用双指针法：一开始使fast为slow的两倍步长 相遇后令fast=headA fast和slow使用相同步长 第一次再次相遇时即为入口
```
//TBC
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        ListNode* slow = headA, *fast = headA, *tmp = headA;
        if(!tmp || !headB) return nullptr;
        while(tmp->next){
            tmp = tmp->next;
        }
        tmp->next = headB;
        slow = headA;
        // the circle's length:c  
        // the length of the other part:l
        // the distance from the entry of the circle: x
        // c + l + x = 2(l+x) => c = l + x => l = c - x
        while(fast->next && fast->next->next){
            slow = slow->next;
            fast = fast->next->next;
            if(fast == slow){
                fast = headA;
                while(fast != slow){
                    fast = fast->next;
                    slow = slow->next;
                }
                tmp->next = nullptr;
                return fast;
            }
        }
        tmp->next = nullptr;
        return nullptr;
    }
};
```
