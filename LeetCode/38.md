//什么鬼题目 第一遍愣是没看懂。。看了discussion照着写的
class Solution {
public:
    string countAndSay(int n) {
        string res("1");
        while(--n){
            string answer = "";
            for(int i = 0; i < res.size(); i++){
                int count = 1;
                while(i < res.size() - 1 && res[i] == res[i+1]){
                    count++; i++;
                }
                answer += (to_string(count) + res[i]);
            }
            res = answer;
        }
        return res;
    }
};
