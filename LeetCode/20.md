class Solution {
public:
    bool isValid(string s) {
        if(s.size() == 0)
            return true;
        string op("([{");
        string cl(")]}");
        stack<short> theStack;
        for(int i = 0; i < s.size(); i++){
            int order = op.find(s[i]);
            if(order != -1)
                theStack.push(order);
            else{
                if(theStack.empty() || theStack.top() != cl.find(s[i]))
                    return false;
                else
                    theStack.pop();
            }
        }
        if(!theStack.empty())
            return false;
        return true;
        
    }
};
