//一个基于垂直扫描思想的算法 8ms
class Solution {
public:
    string longestCommonPrefix(vector<string>& strs) {
        if(strs.size() == 0)
            return "";
        if(strs.size() == 1)
            return strs[0];
        string result;
        int i = 0; 
        bool flag = true;
        for(;flag;i++){
            char tmp = strs[0][i];
            auto iter = strs.begin()+1;
            for(; iter != strs.end(); iter++){
                if(i >= iter->size() || (*iter)[i] != tmp){
                    flag = false;
                    break;
                }
            }
            if(iter == strs.end())
                result += tmp;
        }
        return result;
    }
};
