/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeElements(ListNode* head, int val) {
        ListNode* ptr = new ListNode(-1);
        ptr->next = head;
        head = ptr;
        ptr = head->next;
        ListNode* pre = head;
        while(ptr != NULL){
            if(ptr->val == val){
                pre->next = ptr->next;
                delete(ptr);
                ptr = pre->next;
            }
            else{
                ptr = ptr->next;
                pre = pre->next;
            }
        }
        return head->next;
    }
};
