//一个非常粗暴的算法
//进行了笨拙的边界判断
class Solution {
public:
    int reverse(int x) {
        bool neg = x < 0?true:false;
        int MAX = 214748364;
        if(x == -2147483648) //如果x = -2^32 取负时会越界
            return 0;
        if(neg){
            x = -x;
        }
        int tmp = x % 10;
        int count = 1;
        x = x / 10;
        while(x != 0){
            if(count == 9){ //如果后9位倒置后为2^32的前九位则进行越界判断
                if(tmp > MAX)
                    return 0;
                if(tmp == MAX){
                    if(neg){
                        if(x > 7)
                            return 0;
                    }
                    else{
                        if(x > 8)
                            return 0;
                    }
                }
            }
            tmp = tmp * 10 + x % 10;
            x = x / 10;
            count ++;
        }
        return neg?-tmp:tmp;
    }
};
